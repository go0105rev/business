DROP TABLE USER_INF;
DROP TABLE LAB;
DROP TABLE CONTENT;
DROP TABLE TEAM_INF;
DROP TABLE UNIT_TEST;

CREATE TABLE USER_INF(
    team_id char(5) not null,
    user_id varchar(10) not null,
    password varchar not null,
    user_name char(20) not null,
    gender SMALLINT not null,
    age INTEGER,
    exp INTEGER,
    s_date TIMESTAMP not null,
    e_date TIMESTAMP,
    PRIMARY KEY(user_id)
);

CREATE TABLE LAB(
    scope INTEGER not null,
    scope_name TEXT not null,
    sdate TIMESTAMP,
    acess_flg SMALLINT,
    PRIMARY KEY(scope)
);

CREATE TABLE CONTENT(
    ques_num char(8) not null,
    scope BIGINT,
    ques BIGINT,
    ques_name char(50) not null,
    detail TEXT,
    rule TEXT,
    sdate TIMESTAMP,
    acess_flg SMALLINT,
    PRIMARY KEY(ques_num)
);

CREATE TABLE TEAM_INF(
    team_id char(5) not null,
    team_name char(50) not null,
    s_date TIMESTAMP,
    e_date TIMESTAMP,
    PRIMARY KEY(team_id)
);

CREATE TABLE UNIT_TEST(
    source_id char(25) not null,
    user_id varchar(10) not null,
    ques_num char(8) not null,
    status SMALLINT,
    source BYTEA,
    saveTime TIMESTAMP not null,
    strSaveTime char(21) not null,
    size DOUBLE PRECISION,
    duration DOUBLE PRECISION,
    score BIGINT,
    PRIMARY KEY(source_id)
);
